144 TF createDummyData(int ndigits, int nrofs, uint32_t firstOrbit)
145 {
146  if (nrofs >= ndigits) {
147  throw std::invalid_argument("cannot have more rofs than digits!");
148  }
149  TF tf;
150  tf.digits = createDummyFixedDigits(ndigits);
151  int step = ndigits / nrofs;
152  for (int i = 0; i < nrofs; i++) {
153  o2::InteractionRecord ir(0, firstOrbit + i);
154  tf.rofs.emplace_back(ir, step * i, i == nrofs - 1 ? ndigits - step * i : step);
155  };
156  return tf;
157 }

253 BOOST_DATA_TEST_CASE(CheckReader, digitFileFormats, digitFileFormat)
254 {
255  std::stringstream buffer;
256  writeTestData(buffer, digitFileFormat);
257  DigitReader dr(buffer);
258 
259  std::vector<Digit> digits;
260  std::vector<ROFRecord> rofs;
261 
262  std::array<int, 3> digits_per_tf = {3, 5, 13};
263  for (int itf = 0; itf < 3; ++itf) {
264  dr.read(digits, rofs);
265  BOOST_CHECK_EQUAL(digits.size(), digits_per_tf[itf]);
266  BOOST_CHECK_EQUAL(rofs.size(), 1);
267  BOOST_CHECK_EQUAL(rofs[0].getFirstIdx(), 0);
268  BOOST_CHECK_EQUAL(rofs[0].getLastIdx(), digits_per_tf[itf] - 1);
269  }
270  dr.read(digits, rofs);
271  int ndigits_4th_tf = digitFileFormat.fileVersion == 0 ? 8 : 26;
272  BOOST_CHECK_EQUAL(digits.size(), ndigits_4th_tf);
273  if (digitFileFormat.fileVersion == 0) {
274  BOOST_CHECK_EQUAL(rofs.size(), 1);
275  BOOST_CHECK_EQUAL(rofs[0].getFirstIdx(), 0);
276  BOOST_CHECK_EQUAL(rofs[0].getLastIdx(), 7);
277  } else {
278  BOOST_CHECK_EQUAL(rofs.size(), 3);
279  BOOST_CHECK_EQUAL(rofs[0].getFirstIdx(), 0);
280  BOOST_CHECK_EQUAL(rofs[0].getLastIdx(), 7);
281  BOOST_CHECK_EQUAL(rofs[1].getFirstIdx(), 8);
282  BOOST_CHECK_EQUAL(rofs[1].getLastIdx(), 15);
283  BOOST_CHECK_EQUAL(rofs[2].getFirstIdx(), 16);
284  BOOST_CHECK_EQUAL(rofs[2].getLastIdx(), 25);
285  }
